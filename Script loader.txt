-- Lua Test Script

-- Function to check Lua's basic math
function mathTest()
    local startTime = os.clock()
    local sum = 0
    for i = 1, 1000000 do
        sum = sum + math.sqrt(i)
    end
    local endTime = os.clock()
    return "Math test took: " .. (endTime - startTime) .. " seconds"
end

-- Function to check string operations
function stringTest()
    local str = "Hello, Lua!"
    local reversed = str:reverse()
    return "Reversed string: " .. reversed
end

-- Function to check table (array-like structures)
function tableTest()
    local myTable = {}
    for i = 1, 10000 do
        myTable[i] = i * 2
    end
    return "Table size is: " .. #myTable
end

-- Function to check for error handling
function errorTest()
    local success, err = pcall(function()
        error("This is a test error!")
    end)
    if not success then
        return "Error test result: " .. err
    else
        return "No errors"
    end
end

-- Run the tests
print(mathTest())
print(stringTest())
print(tableTest())
print(errorTest())
